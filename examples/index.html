Circles

basic usage

here the library draw a circle with arc of 90% with default settings of canvas ,with out write the percentage in middle of the circle


new Circle(document.querySelector("#basic-usage"),{
		valueType : PERC,
		maxValue : 100
	},90).startDrawing();

advanced usage

circle with animation and customize the written value 


new Circle(document.querySelector(".addv1"),
	{
		//border width of the circle
		lineForce : 10 ,
		//color of the circle
		color : "gold",
		//to write percentage in middle
		withValue : true,
		//to animate filling circle
		withAnimation : true ,
		//styling percentage
		valueStyle : {
			// set font style
			font:"20px Arial",
			// set color
			color : "red"
		},
		// set angle of start drawing the circle
		startAngle : 46,
		// type of value that you wanna the circle present
		valueType : PERC,
		// max value
		maxValue : 100,
		// animation duration
		animationDuration : 5000
	},90).startDrawing();

circle with filling the rest, customizing value and start drawing when the canvas visible on the page for the first time

by default the animation Duration is 1s 

	var circle = new Circle(document.querySelector(".addv2"),
	{
		//border width of the circle
		lineForce :10 ,
		//color of the circle
		color : "#0093b3",
	    //to write the value in middle
		withValue : true,
		//to draw with animation
		withAnimation : true ,
		//angle of start drawing
		startAngle : 150,
		//styling value
		valueStyle : {
			// color of value
			color : "#0093b3",
			// font of the value
			font:"30px Arial"
		},
		//max angle of the circle
		maxAngle : (240) ,
		//to fill the rest of circle
		fillCirclRest: true,
		//styling the rest of circle
		fillRestStyle : {
			//set color
			color :"#86e1ff"
		},

		valueType : NUMB,
		maxValue : 10000
	},9000);
// START DRAWING WHEN IT'S VISIBLE IN PAGE
document.addEventListener("scroll", function(){
	if(window.scrollY >= window.scrollMaxY - 50 ){
		if(!circle.drawing ){

			circle.startDrawing();
		}
	}
});
